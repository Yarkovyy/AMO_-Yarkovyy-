def naive_match(p, t):
    assert len(p) <= len(t)  # –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ –¥–æ–≤–∂–∏–Ω–∞ —à–∞–±–ª–æ–Ω—É –Ω–µ –±—ñ–ª—å—à–∞ –∑–∞ –¥–æ–≤–∂–∏–Ω—É —Ç–µ–∫—Å—Ç—É
    occurrences = []
    for i in range(0, len(t)-len(p)+1):  # –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ –≤–∏—Ä—ñ–≤–Ω—é–≤–∞–Ω–Ω—è —à–∞–±–ª–æ–Ω—É –Ω–∞ —Ç–µ–∫—Å—Ç
        match = True  # –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ —î –∑–±—ñ–≥, –¥–æ–∫–∏ –Ω–µ –¥–æ–≤–µ–¥–µ–Ω–æ –∑–≤–æ—Ä–æ—Ç–Ω—î
        for j in range(0, len(p)):  # –¥–ª—è –∫–æ–∂–Ω–æ—ó –ø–æ–∑–∏—Ü—ñ—ó —à–∞–±–ª–æ–Ω—É
            if t[i+j] != p[j]:
                match = False  # –ø—Ä–∏–Ω–∞–π–º–Ω—ñ –æ–¥–∏–Ω —Å–∏–º–≤–æ–ª –Ω–µ –∑–±—ñ–≥–∞—î—Ç—å—Å—è
                break
        if match:
            occurrences.append(i)
    return occurrences

def z_func(s):
    # –Ü–Ω—ñ—Ü—ñ–∞–ª—ñ–∑—É—î–º–æ –º–∞—Å–∏–≤ Z, –≤ —è–∫–æ–º—É –±—É–¥–µ–º–æ –∑–±–µ—Ä—ñ–≥–∞—Ç–∏ –∑–Ω–∞—á–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó
    Z = [len(s)] + [0] * len(s)  # –í—Å—Ç–∞–Ω–æ–≤–ª—é—î–º–æ –¥–æ–≤–∂–∏–Ω—É —Ä—è–¥–∫–∞ s —É –ø–µ—Ä—à–æ–º—É –µ–ª–µ–º–µ–Ω—Ç—ñ –º–∞—Å–∏–≤—É Z, –∞ —Ä–µ—à—Ç—É –∑–∞–ø–æ–≤–Ω—é—î–º–æ –Ω—É–ª—è–º–∏
    assert len(s) > 1  # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –¥–æ–≤–∂–∏–Ω–∞ —Ä—è–¥–∫–∞ s –±—ñ–ª—å—à–µ 1
    
    # –ü–æ—á–∞—Ç–∫–æ–≤–∞ –ø–æ—Ä—ñ–≤–Ω—è–Ω–Ω—è s[1:] –∑ –ø—Ä–µ—Ñ—ñ–∫—Å–æ–º
    for i in range(1, len(s)):
        # –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å–∏–º–≤–æ–ª–∏ s[i] —Ç–∞ s[i-1]. –Ø–∫—â–æ –≤–æ–Ω–∏ –æ–¥–Ω–∞–∫–æ–≤—ñ, –∑–±—ñ–ª—å—à—É—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è Z[1]
        if s[i] == s[i - 1]:
            Z[1] += 1
        else:
            break  # –Ø–∫—â–æ –∑–Ω–∞–π–¥–µ–Ω–∏–π —Å–∏–º–≤–æ–ª –Ω–µ –∑–±—ñ–≥–∞—î—Ç—å—Å—è –∑ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ–º, –ø—Ä–∏–ø–∏–Ω—è—î–º–æ –ø—Ä–æ—Ü–µ—Å
    
    # –£—Å—Ç–∞–Ω–æ–≤–ª—é—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è –∑–º—ñ–Ω–Ω–∏—Ö r —ñ l
    r, l = 0, 0
    if Z[1] > 0:
        r, l = Z[1], 1
    
    # –û–±—á–∏—Å–ª–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ k –≤—ñ–¥ 2 –¥–æ –¥–æ–≤–∂–∏–Ω–∏ —Ä—è–¥–∫–∞ s
    for k in range(2, len(s)):
        assert Z[k] == 0  # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞, —â–æ –∑–Ω–∞—á–µ–Ω–Ω—è Z[k] —Å–ø–æ—á–∞—Ç–∫—É –¥–æ—Ä—ñ–≤–Ω—é—î 0
        if k > r:
            # –í–∏–ø–∞–¥–æ–∫ 1: k –∑–∞ –º–µ–∂–∞–º–∏ –ø–æ—Ç–æ—á–Ω–æ–≥–æ –¥—ñ–∞–ø–∞–∑–æ–Ω—É [l, r]
            for i in range(k, len(s)):
                # –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å–∏–º–≤–æ–ª–∏ s[i] —Ç–∞ s[i-k]. –Ø–∫—â–æ –≤–æ–Ω–∏ –∑–±—ñ–≥–∞—é—Ç—å—Å—è, –∑–±—ñ–ª—å—à—É—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è Z[k]
                if s[i] == s[i - k]:
                    Z[k] += 1
                else:
                    break
            # –û–Ω–æ–≤–ª—é—î–º–æ –∑–Ω–∞—á–µ–Ω–Ω—è –∑–º—ñ–Ω–Ω–∏—Ö r —Ç–∞ l
            r, l = k + Z[k] - 1, k
        else:
            # –í–∏–ø–∞–¥–æ–∫ 2: k –∑–Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ –ø–æ—Ç–æ—á–Ω–æ–≥–æ –¥—ñ–∞–ø–∞–∑–æ–Ω—É [l, r]
            # –û–±—á–∏—Å–ª—é—î–º–æ –¥–æ–≤–∂–∏–Ω—É –±–µ—Ç–∞ (nbeta) —Ç–∞ –∑–Ω–∞—á–µ–Ω–Ω—è Z-—Ñ—É–Ω–∫—Ü—ñ—ó –Ω–∞ –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ–º—É –∫—Ä–æ—Ü—ñ (Zkp)
            nbeta = r - k + 1
            Zkp = Z[k - l]
            if nbeta > Zkp:
                # –ü—ñ–¥–≤–∏–ø–∞–¥–æ–∫ 2–∞: Zkp –≤–∏–≥—Ä–∞—î
                Z[k] = Zkp
            else:
                # –ü—ñ–¥–≤–∏–ø–∞–¥–æ–∫ 2–±: –ü–æ—Ä—ñ–≤–Ω—é—î–º–æ —Å–∏–º–≤–æ–ª–∏, —â–æ –∑–Ω–∞—Ö–æ–¥—è—Ç—å—Å—è –±–µ–∑–ø–æ—Å–µ—Ä–µ–¥–Ω—å–æ –ø—ñ—Å–ª—è r
                nmatch = 0
                for i in range(r + 1, len(s)):
                    if s[i] == s[i - k]:
                        nmatch += 1
                    else:
                        break
                l, r = k, r + nmatch
                Z[k] = r - k + 1
    return Z

def zMatch(p, t):

    # –ü–æ—á–∏–Ω–∞—î–º–æ –∑ –ø–æ–∑–∏—Ü—ñ—ó, —â–æ –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—î –∫—ñ–Ω—Ü—é —à–∞–±–ª–æ–Ω—É (len(p) + 1) —ñ –¥–æ –∫—ñ–Ω—Ü—è —Ä—è–¥–∫–∞ s.
    for i in range(len(p) + 1, len(s)):

def compress_with_z(s):

    for i in range(1, len(s)):
        # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –≤–∏–∫–æ–Ω—É—é—Ç—å—Å—è —É–º–æ–≤–∏ –¥–ª—è –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ–≥–æ —Å—Ç–∏—Å–Ω–µ–Ω–Ω—è

        return s
