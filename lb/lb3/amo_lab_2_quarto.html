def insertion_sort(nums):
    # –ø–æ—á–∏–Ω–∞—î–º–æ –∑ –¥—Ä—É–≥–æ–≥–æ –µ–ª–µ–º–µ–Ω—Ç
    for j in range(1, len(nums)):

        # –ó–±–µ—Ä—ñ–≥–∞—î–º–æ –ø–æ—Å–∏–ª–∞–Ω–Ω—è –Ω–∞ —ñ–Ω–¥–µ–∫—Å –ø–µ—Ä—à–æ–≥–æ –µ–ª–µ–º–µ–Ω—Ç–∞
        i = j - 1
        # –ï–ª–µ–º–µ–Ω—Ç–∏ –≤—ñ–¥—Å–æ—Ä—Ç–æ–≤–∞–Ω–æ–≥–æ —Å–µ–≥—Ä–º–µ–Ω—Ç–∞ –ø–µ—Ä–µ–º—ñ—â—É—î–º–æ –≤–ø–µ—Ä–µ–¥, —è–∫—â–æ –≤–æ–Ω–∏ –±—ñ–ª—å—à–µ –µ–ª–µ–º–µ–Ω—Ç–∞ –≤—Å—Ç–∞–≤–∫–∏
        while i >= 0 and nums[i] > key:
            nums[i + 1] = nums[i]
            i -= 1
        # –í—Å—Ç–∞–≤–ª—è—î–º–æ –µ–ª–µ–º–µ–Ω—Ç
        nums[i + 1] = key

import random
from array import array
import time

def bubble_sort(array):
    length = len(array)
    for i in range(0, length):
        for j in range(0, length - i - 1):
            if array[j] > array[j + 1]:
                temp = array[j]
                array[j] = array[j + 1]
                array[j + 1] = temp
